c
#include <stdio.h>
#include <math.h>
#include <string.h>

#define ROWS    20
#define COLS    50
#define MAX_OBJ 50

/* Shape types */
#define SHAPE_CIRCLE   1
#define SHAPE_RECT     2
#define SHAPE_LINE     3
#define SHAPE_TRIANGLE 4

/* 2D canvas array */
char canvas[ROWS][COLS];

/* Object descriptor */
typedef struct {
    int type;
    int x, y, r, w, h;
    int x2, y2;
    int active;
} Object;

Object objects[MAX_OBJ];
int obj_count = 0;

/* ─── Canvas functions ─── */
void clear_canvas() {
    for (int i = 0; i < ROWS; i++)
        for (int j = 0; j < COLS; j++)
            canvas[i][j] = '_';
}

void plot(int row, int col) {
    if (row >= 0 && row < ROWS && col >= 0 && col < COLS)
        canvas[row][col] = '*';
}

void display_canvas() {
    printf("\n");
    for (int i = 0; i < ROWS; i++) {
        for (int j = 0; j < COLS; j++)
            printf("%c", canvas[i][j]);
        printf("\n");
    }
    printf("\n");
}

/* ─── Shape functions ─── */
void draw_line(int x0, int y0, int x1, int y1) {
    int dx = abs(x1 - x0), dy = abs(y1 - y0);
    int sx = (x0 < x1) ? 1 : -1;
    int sy = (y0 < y1) ? 1 : -1;
    int err = dx - dy;
    while (1) {
        plot(y0, x0);
        if (x0 == x1 && y0 == y1) break;
        int e2 = 2 * err;
        if (e2 > -dy) { err -= dy; x0 += sx; }
        if (e2 <  dx) { err += dx; y0 += sy; }
    }
}

void draw_rectangle(int x, int y, int w, int h) {
    draw_line(x,     y,     x + w, y    );
    draw_line(x,     y + h, x + w, y + h);
    draw_line(x,     y,     x,     y + h);
    draw_line(x + w, y,     x + w, y + h);
}

void draw_circle(int cx, int cy, int r) {
    int x = 0, y = r, d = 1 - r;
    while (x <= y) {
        plot(cy + y, cx + x); plot(cy - y, cx + x);
        plot(cy + y, cx - x); plot(cy - y, cx - x);
        plot(cy + x, cx + y); plot(cy - x, cx + y);
        plot(cy + x, cx - y); plot(cy - x, cx - y);
        if (d < 0) d += 2 * x + 3;
        else { d += 2 * (x - y) + 5; y--; }
        x++;
    }
}

void draw_triangle(int x0, int y0, int x1, int y1, int x2, int y2) {
    draw_line(x0, y0, x1, y1);
    draw_line(x1, y1, x2, y2);
    draw_line(x2, y2, x0, y0);
}

/* ─── Render all objects ─── */
void render_all() {
    clear_canvas();
    for (int i = 0; i < obj_count; i++) {
        Object *o = &objects[i];
        if (!o->active) continue;
        switch (o->type) {
            case SHAPE_CIRCLE:   draw_circle(o->x, o->y, o->r); break;
            case SHAPE_RECT:     draw_rectangle(o->x, o->y, o->w, o->h); break;
            case SHAPE_LINE:     draw_line(o->x, o->y, o->w, o->h); break;
            case SHAPE_TRIANGLE: draw_triangle(o->x, o->y, o->w, o->h, o->x2, o->y2); break;
        }
    }
}

/* ─── Add object ─── */
void add_object() {
    if (obj_count >= MAX_OBJ) {
        printf("Canvas full!\n");
        return;
    }
    Object o;
    memset(&o, 0, sizeof(o));
    o.active = 1;

    printf("\nSelect shape:\n");
    printf("1. Circle\n");
    printf("2. Rectangle\n");
    printf("3. Line\n");
    printf("4. Triangle\n");
    printf("Choice: ");
    scanf("%d", &o.type);

    switch (o.type) {
        case SHAPE_CIRCLE:
            printf("Centre X (0-49): "); scanf("%d", &o.x);
            printf("Centre Y (0-19): "); scanf("%d", &o.y);
            printf("Radius: ");          scanf("%d", &o.r);
            break;
        case SHAPE_RECT:
            printf("Top-left X: "); scanf("%d", &o.x);
            printf("Top-left Y: "); scanf("%d", &o.y);
            printf("Width: ");      scanf("%d", &o.w);
            printf("Height: ");     scanf("%d", &o.h);
            break;
        case SHAPE_LINE:
            printf("Start X1: "); scanf("%d", &o.x);

            printf("End X2: ");   scanf("%d", &o.w);
            printf("End Y2: ");   scanf("%d", &o.h);
            break;
        case SHAPE_TRIANGLE:
            printf("Vertex A X: "); scanf("%d", &o.x);

            printf("Vertex C Y: "); scanf("%d", &o.y2);
            break;
        default:
            printf("Invalid choice!\n");

    objects[obj_count++] = o;
    printf("Shape added successfully!\n");
    render_all();
    display_canvas();

/* ─── List objects ─── */
void list_objects() {
    printf("\n%-4s %-10s %-20s\n", "ID", "Type", "Parameters");
    printf("%-4s %-10s %-20s\n", "----", "----------", "--------------------");
    int found = 0;
    for (int i = 0; i < obj_count; i++) {
        Object *o = &objects[i];
        if (!o->active) continue;
        found = 1;
        const char *tn = o->type == SHAPE_CIRCLE   ? "Circle" :
                         o->type == SHAPE_RECT      ? "Rectangle" :
                         o->type == SHAPE_LINE      ? "Line" : "Triangle";
        printf("%-4d %-10s ", i, tn);
        if (o->type == SHAPE_CIRCLE)
            printf("cx=%d cy=%d r=%d", o->x, o->y, o->r);
        else if (o->type == SHAPE_RECT)
            printf("x=%d y=%d w=%d h=%d", o->x, o->y, o->w, o->h);
        else if (o->type == SHAPE_LINE)
            printf("(%d,%d)->(%d,%d)", o->x, o->y, o->w, o->h);
        else
            printf("A(%d,%d) B(%d,%d) C(%d,%d)", o->x, o->y, o->w, o->h, o->x2, o->y2);
        printf("\n");
    }
    if (!found)
        printf("  (no objects yet)\n");
    printf("\n");
}

/* ─── Delete object ─── */
void delete_object() {
    list_objects();
    int id;
    printf("Enter ID to delete: ");
    scanf("%d", &id);
    if (id >= 0 && id < obj_count && objects[id].active) {
        objects[id].active = 0;
        printf("Object deleted!\n");
        render_all();
        display_canvas();
    } else {
        printf("Invalid ID!\n");
    }
}

    list_objects();

            printf("Vertex B X: "); scanf("%d", &o->w);

            printf("Vertex C X: "); scanf("%d", &o->x2);
            printf("Vertex C Y: "); scanf("%d", &o->y2);

    }

    display_canvas();

/* ═══════════════════════════════════════════
   Drawing primitives
═══════════════════════════════════════════ */
void draw_line(int x0, int y0, int x1, int y1) {
    int dx = abs(x1-x0), dy = abs(y1-y0);
    int sx = (x0<x1)?1:-1, sy = (y0<y1)?1:-1;
    int err = dx - dy;
    while(1) {
        plot(y0, x0);
        if(x0==x1 && y0==y1) break;
        int e2 = 2*err;
        if(e2 > -dy){ err -= dy; x0 += sx; }

}

    for(int i=0;i<ROWS;i++){
        for(int j=0;j<COLS;j++){

            if(c=='*'){

/* ═══════════════════════════════════════════
   Get integer input via status bar
═══════════════════════════════════════════ */
int get_int(const char *prompt) {
    werase(win_status);
    wattron(win_status, COLOR_PAIR(CLR_STATUS) | A_BOLD);
    mvwprintw(win_status, 0, 1, " >> %s: ", prompt);
    wattroff(win_status, COLOR_PAIR(CLR_STATUS) | A_BOLD);

    int row = 4;

                         o->type==SHAPE_LINE?"Line":"Triangle";

            mvwprintw(win_menu, row, 2, "%-4d %-10s x=%-2d y=%-2d w=%-2d h=%-2d", i,tn,o->x,o->y,o->w,o->h);
        else if(o->type==SHAPE_LINE)
            mvwprintw(win_menu, row, 2, "%-4d %-10s (%d,%d)->(%d,%d)", i,tn,o->x,o->y,o->w,o->h);

            mvwprintw(win_menu, row, 2, "%-4d %-10s 3 pts", i,tn);

    mvwprintw(win_menu, ROWS-1, 2, " Press any key to go back ");

/* ═══════════════════════════════════════════
   Add object menu
═══════════════════════════════════════════ */
void menu_add(void) {
    const char *items[] = {
        "1. Circle",
        "2. Rectangle",
        "3. Line",
        "4. Triangle",
        "0. Back"
    };
    int n = 5, sel = 0;

    while(1){
        draw_menu(sel, items, n, "ADD SHAPE");
        int ch = wgetch(win_menu);
        if(ch == KEY_UP)   sel = (sel-1+n)%n;
        if(ch == KEY_DOWN) sel = (sel+1)%n;
        if(ch == '\n' || ch == KEY_ENTER || ch == 13){
            if(sel==4) return;
            Object o; memset(&o,0,sizeof o);
            o.type = sel+1; o.active = 1;

            switch(o.type){
                case SHAPE_CIRCLE:
                    o.x = get_int("Circle centre X (0-49)");
                    o.y = get_int("Circle centre Y (0-19)");
                    o.r = get_int("Radius");
                    break;
                case SHAPE_RECT:
                    o.x = get_int("Rectangle top-left X");
                    o.y = get_int("Rectangle top-left Y");
                    o.w = get_int("Width");
                    o.h = get_int("Height");
                    break;
                case SHAPE_LINE:
                    o.x = get_int("Line start X1");
                    o.y = get_int("Line start Y1");
                    o.w = get_int("Line end X2");
                    o.h = get_int("Line end Y2");
                    break;
                case SHAPE_TRIANGLE:
                    o.x  = get_int("Vertex A X");
                    o.y  = get_int("Vertex A Y");
                    o.w  = get_int("Vertex B X");
                    o.h  = get_int("Vertex B Y");
                    o.x2 = get_int("Vertex C X");
                    o.y2 = get_int("Vertex C Y");
                    break;
            }
            if(obj_count < MAX_OBJ){
                objects[obj_count++] = o;
                set_status("Shape added successfully!");
            } else {
                set_status("Canvas full!");
            }
            display_canvas();
            return;
        }
    }
}

void main_menu(void) {
    const char *items[] = {
        "1. Add Object",
        "2. Delete Object",
        "3. Modify Object",
        "4. List Objects",
        "5. Clear Canvas",
        "0. Exit"
    };
    int n=6, sel=0;

    while(1){
        draw_menu(sel, items, n, "MAIN MENU");
        display_canvas();

        if(ch==KEY_UP)   sel=(sel-1+n)%n;
        if(ch==KEY_DOWN) sel=(sel+1)%n;

                case 1: menu_delete(); break;

                    display_canvas();
                    break;
                case 5: return;
            }
        }
    }

20 rows and 50 columns
Empty cells hold ' ' (space), drawn cells hold '*'
clear_canvas() fills everything with spaces
plot(row, col) puts a * at a specific position
2. The Four Shape Functions

draw_line() — uses Bresenham's algorithm which is a mathematical way to draw a straight line between two points on a grid
draw_rectangle() — simply calls draw_line() 4 times for each side
draw_circle() — uses midpoint circle algorithm which plots 8 symmetric points at a time around the center
draw_triangle() — calls draw_line() 3 times connecting 3 vertices
3. Object Management

Every shape you add is stored as an Object struct in an array objects[50]
Each object stores its type, coordinates, and an active flag
Add — fills a new Object and adds it to the array
Delete — sets active = 0 so it gets skipped during drawing
Modify — finds the object by ID and updates its values
render_all() loops through all active objects and redraws them every time something changes
4. Display Function

display_canvas() loops through the 2D array and prints each character on screen using PDCurses' mvwaddch()
* is printed in green bold
Background is printed as _ in white

This is an important point your teacher may ask. Say this:

"ncurses is a Linux library and doesn't work on Windows. PDCurses is the Windows equivalent — it has the same functions and works exactly the same way, just compatible with Windows terminals."

Built using PDCurses windows — win_title, win_menu, win_canvas, win_status
Arrow keys move the highlight up and down

Sure! Here's everything explained simply.

What is MSYS2?

MSYS2 is a software platform for Windows that gives you Linux-like tools and a terminal. Think of it as a toolbox that brings developer tools to Windows that normally only exist on Linux.

You download it from msys2.org and install it like any normal Windows application.

What is MinGW64?

MinGW64 stands for Minimalist GNU for Windows 64-bit.

It is a compiler toolchain that comes inside MSYS2. It lets you compile and run C programs on Windows using the same tools that Linux developers use.

When you open MSYS2 MinGW64 terminal — that's the environment where MinGW64 is active and ready to use.

What is GCC?

GCC stands for GNU Compiler Collection.

It is the actual compiler — the tool that converts your C code into an executable .exe file that your computer can run.

You installed it with:

bash
pacman -S mingw-w64-x86_64-gcc
And verified it with:

bash
gcc --version
# gcc.exe (Rev13, Built by MSYS2 project) 15.2.0
What is pacman?

pacman is the package manager of MSYS2 — like an app store for the terminal. Whenever you needed to install something you used:

bash
pacman -S package-name
How they all connect:

MSYS2
  └── gives you a Linux-like environment on Windows
        └── MinGW64
              └── gives you the compiler toolchain
                    └── GCC
                          └── compiles your C code into .exe
                                └── PDCurses
                                      └── gives you colored terminal UI
What is PDCurses?

PDCurses stands for Public Domain Curses.

On Linux, developers use a library called ncurses to make terminal menus, colors, and windows

So on Windows we use PDCurses which is the Windows replacement
It has the exact same functions as ncurses — initscr(), newwin(), mvwprintw() etc.
